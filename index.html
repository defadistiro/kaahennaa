const FOLDER_NAME = 'KaahennaaBackup';

function getOrCreateFolder() {
  const folders = DriveApp.getFoldersByName(FOLDER_NAME);
  return folders.hasNext() ? folders.next() : DriveApp.createFolder(FOLDER_NAME);
}

// ── Merge dua array berdasarkan field id/nama ─────────────────────────
function mergeArray(oldArr, newArr, idField) {
  if (!oldArr || oldArr.length === 0) return newArr || [];
  if (!newArr || newArr.length === 0) return oldArr;
  const map = {};
  oldArr.forEach(function(item) {
    const key = item[idField];
    if (key !== undefined && key !== null && key !== '') map[String(key)] = item;
  });
  newArr.forEach(function(item) {
    const key = item[idField];
    if (key !== undefined && key !== null && key !== '') map[String(key)] = item;
    else map['__nokey_' + Math.random()] = item;
  });
  return Object.values(map);
}

// ── Merge dua payload all.json ────────────────────────────────────────
function mergePayload(oldData, newData) {
  var result = JSON.parse(JSON.stringify(newData)); // shallow clone baru
  // Merge setiap array data dengan id yang tepat per key
  var idFields = {
    bookings:                  'id',
    masterHenna:               'id',
    masterVendor:              'id',
    masterKuku:                'id',
    stokData:                  'id',
    pengeluaran:               'id',
    masterKategoriPengeluaran: 'id'
  };
  Object.keys(idFields).forEach(function(key) {
    if (oldData[key] || newData[key]) {
      result[key] = mergeArray(oldData[key] || [], newData[key] || [], idFields[key]);
    }
  });
  // Perbarui _meta berdasarkan hasil merge
  result._meta = {
    exportedAt: new Date().toISOString(),
    source: 'kaahennaa-merged',
    mergedAt: new Date().toISOString(),
    summary: {
      jenisHenna: (result.masterHenna  || []).length,
      vendor:     (result.masterVendor || []).length,
      booking:    (result.bookings     || []).length,
      kuku:       (result.masterKuku   || []).length,
      stok:       (result.stokData     || []).length,
      pengeluaran:(result.pengeluaran  || []).length,
      kategoriPengeluaran: (result.masterKategoriPengeluaran || []).length
    }
  };
  return result;
}

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const folder = getOrCreateFolder();

    if (data.action === 'test') {
      return json({ success: true,
        message: 'Koneksi OK! Folder: ' + folder.getName(),
        email: Session.getActiveUser().getEmail() });
    }

    if (data.action === 'push') {
      const filename = (data.key || 'all') + '.json';
      var newPayload  = data.payload;

      // ── MERGE: baca data lama dulu, gabung dengan data baru ──────────
      const existing = folder.getFilesByName(filename);
      var oldFile = null;
      if (existing.hasNext()) oldFile = existing.next();

      if (oldFile) {
        try {
          const oldContent = oldFile.getBlob().getDataAsString();
          const oldPayload = JSON.parse(oldContent);
          newPayload = mergePayload(oldPayload, newPayload);
        } catch(mergeErr) {
          // Jika parse gagal, lanjut pakai newPayload saja
        }
        // Ganti isi file lama (update in-place, TIDAK hapus & buat baru)
        oldFile.setContent(JSON.stringify(newPayload, null, 2));
        return json({
          success: true,
          fileId: oldFile.getId(),
          filename: filename,
          size: JSON.stringify(newPayload).length,
          merged: true
        });
      } else {
        // Belum ada file → buat baru
        const file = folder.createFile(filename, JSON.stringify(newPayload, null, 2), MimeType.PLAIN_TEXT);
        return json({ success: true, fileId: file.getId(),
          filename: filename, size: JSON.stringify(newPayload).length, merged: false });
      }
    }

    if (data.action === 'pull') {
      const filename = (data.key || 'all') + '.json';
      const files = folder.getFilesByName(filename);
      if (!files.hasNext()) {
        return json({ success: false, error: 'File ' + filename + ' belum ada di Drive' });
      }
      const file = files.next();
      const content = file.getBlob().getDataAsString();
      const parsed = JSON.parse(content);
      return json({
        success: true,
        payload: parsed,
        lastModified: file.getLastUpdated().toISOString(),
        fileId: file.getId()
      });
    }

    return json({ success: false, error: 'Action tidak dikenal: ' + data.action });

  } catch(err) {
    return json({ success: false, error: err.toString() });
  }
}

function doGet(e) {
  return json({ status: 'KaaHennaa Cloud Bridge OK' });
}

function json(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
